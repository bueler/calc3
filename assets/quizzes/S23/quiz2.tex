\problem{4}  Compute $\bu\times \bv$ if $\bu=2\bi+3\bj$ and $\bv= \bj+2\bk$.  % 2.4 # 185
\vfill

\problem{4}  Suppose $P(0,0,0)$ is a point in the plane with normal vector $\bn=\left<-3, 2, -1\right>$.  Find the general equation of the plane.    Express your answer in the form $ax+by+cz+d=0$.  % 2.5 #270
\vfill

\clearpage\newpage
\problem{4}  
Find a general equation of the plane through the three points $P(3,-1,2)$, $Q(1,0,1)$, and $R(0,-1,1)$.  Express your answer in the form $ax+by+cz+d=0$. % like 2.5 # 281
\vfill


\problem{5}  Consider the line passing through the two points $P(4,0,5)$ and $Q(2,3,1)$.  % 2.5 # 244

\begin{subproblems}
\item Find a vector equation of the line.
\vspace{2.2in}

\item Find parametric equations of the line.
\vspace{1.2in}
\end{subproblems}
