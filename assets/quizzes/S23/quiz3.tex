\problem{5}  Suppose {\large\, $\br(t) = \left<t,\cos t,\sin t\right>$}.

\begin{subproblems}
\item Sketch this three-dimensional (space) curve.
\vspace{2.7in}

\item Is $\br(t)$ continuous at $t=\pi$?  If so, explain why in a few words, and state as an equation.
\vspace{1.0in}
\end{subproblems}

\problem{4}  Consider the vector-valued function {\large\, $\br(t) = t^2\, \bi + \sqrt{t-3} \,\bj + \frac{5}{2t-1} \,\bk$}.

\begin{subproblems}
\item What is the domain of {\large $\br(t)$}?
\vspace{1.2in}

\item Compute {\large\, $\ds \lim_{t\to 3^+} \br(t)$}.
\vfill
\end{subproblems}

\vfill

\vfill

\problem{8}  The vector-valued function {\large\,$\br(t) = a \sin(\omega\, t)\,\bi + a \cos(\omega\, t)\,\bj$}, for constants $a>0$ and $\omega>0$, describes uniform circular motion.

\begin{subproblems}
\item Show that the velocity {\large $\bv(t)$} at any time is orthogonal to the position {\large $\br(t)$}.
\vfill

\item Show that the acceleration {\large $\ba(t)$} at any time is orthogonal to the velocity {\large $\bv(t)$}.
\vfill
\end{subproblems}
