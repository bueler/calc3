\problem{8}  The volume of a right circular cone is \,{\large $V=\frac{1}{3} \pi r^2 h$}.

\begin{subproblems}
\item Find the differential  \,{\large $dV$}.
\vspace{2.5in}

\item A machine makes cones for ice cream, with target values $r=3$ cm and $h=10$ cm, thus a target volume of $V = 30\pi \,\text{cm}^3$.  However, the machine is only accurate to within $1$ cm in $r$ and $h$.  Use the differential to estimate the maximum deviation in volume away from the target volume.
\vfill
\end{subproblems}
